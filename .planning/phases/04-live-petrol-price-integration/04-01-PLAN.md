**API: Fair Fuel Open Data API (Service Victoria)**

Endpoint: `GET https://api.fuel.service.vic.gov.au/open-data/v1/fuel/prices`

Required headers (from API docs v1.0):
- `x-consumer-id`: Consumer ID from `VITE_SERVO_SAVER_CONSUMER_ID`
- `x-transactionid`: UUID v4, unique per request — use `crypto.randomUUID()`
- `User-Agent`: `'Routed/1.0'`

Response shape (prices in **cents per litre**):
```json
{
  "fuelPriceDetails": [
    {
      "fuelStation": { "address": "123 Main St, Melbourne VIC 3000", ... },
      "fuelPrices": [
        { "fuelType": "U91", "price": 188.8, "isAvailable": true },
        { "fuelType": "DSL", "price": 198.0, "isAvailable": true }
      ]
    }
  ]
}
```

Full implementation of `tryFetchServoSaverPrices`:
```typescript
async function tryFetchServoSaverPrices(consumerId: string): Promise<{ petrol: number; diesel: number } | null> {
  const res = await fetch('https://api.fuel.service.vic.gov.au/open-data/v1/fuel/prices', {
    headers: {
      'x-consumer-id': consumerId,
      'x-transactionid': crypto.randomUUID(),
      'User-Agent': 'Routed/1.0',
    },
  })
  if (!res.ok) return null

  const data = await res.json() as {
    fuelPriceDetails: Array<{
      fuelPrices: Array<{ fuelType: string; price: number; isAvailable: boolean }>
    }>
  }

  const petrolCents: number[] = []
  const dieselCents: number[] = []

  for (const station of data.fuelPriceDetails) {
    for (const fp of station.fuelPrices) {
      if (!fp.isAvailable || typeof fp.price !== 'number' || fp.price <= 0) continue
      if (fp.fuelType === 'U91') petrolCents.push(fp.price)
      if (fp.fuelType === 'DSL') dieselCents.push(fp.price)
    }
  }

  if (petrolCents.length === 0 || dieselCents.length === 0) return null

  const avg = (arr: number[]) => arr.reduce((a, b) => a + b, 0) / arr.length
  // API returns cents/L — convert to $/L
  return {
    petrol: avg(petrolCents) / 100,
    diesel: avg(dieselCents) / 100,
  }
}
```

Notes:
- Averages all available VIC stations (dataset is VIC-only, so no Melbourne filtering needed)
- `U91` = Unleaded 91 (standard petrol). `DSL` = Diesel.
- Rate limit: 10 req/60s — the 24h Firestore cache means we hit this at most once per user per day
